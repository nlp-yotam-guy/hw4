#    S  = sentence
#    NP = noun phrase
#    VP = TVerb phrase
#    PP = prepositional phrase
#    And = link conjunctions
#    That = sentence conjunctions
#    Det = determiner (sometimes called "article")
#    Prep = preposition
#    Adj = adjective
#    IVerb = intransitive TVerb
#    TVerb = transitive TVerb
#    VerbThat = verb that ('thought that')
#    Vbz = is
